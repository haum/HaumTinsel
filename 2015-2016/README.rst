Create virtualenv with django:

	user@host:$ virtualenv2 .haumtinsel2 -ppython2
	
	user@host:$ source .haumtinsel2/bin/activate
	
	user@host:$ pip install django

Run server:

	user@host:$ source .haumtinsel2/bin/activate
	
	user@host:$ cd django_website
	
	user@host:$ ./manage.py runserver
